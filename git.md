```
git init - инициализация пустого репозитория
git add - добавить в индекс
git commit - сделать снимок (коммит) изменений
    [-a --all] - добавляет все файлы перед коммитом (git add (которые уже добавлены в индекс) + git commit)
    [-m --message] - написать сообщение коммита после комманды: git commit -m 'message'
    [--amend] - редактирование сообщения последнего коммита: git commit --amend
git checkout - переключение на ветку или коммит, отмена изменения файла (git checkout branch)
git log - посмотреть историю коммитов
    [-p --patch] - показывает внёсенные изменения построчно
    [-- path/to/file] - выводит все коммиты которые затронули path/to/file
    [--oneline] - выводит коммитов построчно
    [--graph] - строит граф ASKII символами
    [--all] - выводит историю коммитов в других ветках
git show - показывает изменения в коммите (git show file)
git rebase - версия слияния (переносит коммиты с одной ветки на другую (коммит поверх))
git merge - версия слияния (создается новый общий коммит при слиянии)
git stash - спрятать изменения без добавления в индекс (git stash apply - вернуть эти изменения)
git reset HEAD~n - откатиться на n коммитов назад (при этом те, которые выше, перетираются)
git restore filename - откатывает внесённые изменения до предыдущего коммита
    [--staged] - Удаление файла из состояния "добавленного к коммиту
git branch - работа с ветками
git remote - работа с удалёнными репозиториями
git push - отправить изменения
    [-u --set-upstream] - создаёт отслеживаемую ветку в удаленном репозитории
    [-f --force] - принудительно отправить
git fetch - стягивает изменения без слияния
```
